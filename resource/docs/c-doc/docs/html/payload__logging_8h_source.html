92  break;

94  // Note: Only prints the URI (if available), to print further, you'll

98  break;
99  case OCREP_PROP_ARRAY:
100  switch(val->arr.type)
101  {
102  case OCREP_PROP_INT:

184 INLINE_API void OCPayloadLogRep(LogLevel level, OCRepPayload* payload)
185 {
186  OIC_LOG(level, (PL_TAG), "Payload Type: Representation");
187  uint32_t i = 1;
188  for (OCRepPayload* rep = payload; rep; rep = rep->next, ++i)
189  {
190  OIC_LOG_V(level, PL_TAG, "\tResource #%d", i);
191  if (rep->uri)
192  {
193  OIC_LOG_V(level, PL_TAG, "\tURI:%s", rep->uri);
194  }
195  if (rep->types)
196  {
197  OIC_LOG(level, PL_TAG, "\tResource Types:");
198  for (OCStringLL* strll = rep->types; strll; strll = strll->next)
199  {
200  OIC_LOG_V(level, PL_TAG, "\t\t%s", strll->value);
201  }
202  }
203  if (rep->interfaces)
204  {
205  OIC_LOG(level, PL_TAG, "\tInterfaces:");
206  for (OCStringLL* strll = rep->interfaces; strll; strll = strll->next)
207  {
208  OIC_LOG_V(level, PL_TAG, "\t\t%s", strll->value);
209  }
210  }
211  OIC_LOG(level, PL_TAG, "\tValues:");
212  OCPayloadLogRepValues(level, rep->values);
213  }

224 INLINE_API void OCPayloadLogDiscovery(LogLevel level, OCDiscoveryPayload* payload)
225 {
226  OIC_LOG(level, PL_TAG, "Payload Type: Discovery");
227 
228  while(payload && payload->resources)
229  {
230  OIC_LOG_V(level, PL_TAG, "\tDI: %s", payload->sid);
231  if (payload->name)
232  {
233  OIC_LOG_V(level, PL_TAG, "\tNAME: %s", payload->name);
234  }
235 
236  if (payload->type)
237  {
238  OIC_LOG(level, PL_TAG, "\tResource Type:");
239  OCStringLLPrint(level, payload->type);
240  }
241 
242  if (payload->iface)
243  {
244  OIC_LOG(level, PL_TAG, "\tInterface:");
245  OCStringLLPrint(level, payload->iface);
246  }
247 
248  OCResourcePayload* res = payload->resources;
249 
250  uint32_t i = 1;
251  while(res)
252  {
253  OIC_LOG_V(level, PL_TAG, "\tLink#%d", i);
254  OIC_LOG_V(level, PL_TAG, "\tURI:%s", res->uri);
255  if (res->rel)
256  {
257  OIC_LOG_V(level, PL_TAG, "\tRelation:%s", res->rel);
258  }
259  if (res->anchor)
260  {
261  OIC_LOG_V(level, PL_TAG, "\tAnchor:%s", res->anchor);
262  }
263  OIC_LOG(level, PL_TAG, "\tResource Types:");
264  OCStringLL* strll = res->types;
265  while(strll)
266  {
267  OIC_LOG_V(level, PL_TAG, "\t\t%s", strll->value);
268  strll = strll->next;
269  }
270  OIC_LOG(level, PL_TAG, "\tInterfaces:");
271  strll = res->interfaces;
272  while(strll)
273  {
274  OIC_LOG_V(level, PL_TAG, "\t\t%s", strll->value);
275  strll = strll->next;
276  }
277 
278  OIC_LOG_V(level, PL_TAG, "\tBitmap: %u", res->bitmap);
279  OIC_LOG_V(level, PL_TAG, "\tSecure?: %s", res->secure ? "true" : "false");
280  OIC_LOG_V(level, PL_TAG, "\tPort: %u", res->port);
281 
282  uint32_t j = 1;
283  OCEndpointPayload* eps = res->eps;
284  while (eps)
285  {
286  OIC_LOG_V(level, PL_TAG, "\tEndpoint #%d", j);
287  OIC_LOG_V(level, PL_TAG, "\t\ttps: %s", eps->tps);
288  OIC_LOG_V(level, PL_TAG, "\t\taddr: %s", eps->addr);
289  OIC_LOG_V(level, PL_TAG, "\t\tport: %d", eps->port);
290  OIC_LOG_V(level, PL_TAG, "\t\tpri: %d", eps->pri);
291  eps = eps->next;
292  ++j;
293  }
294 
295  OIC_LOG(level, PL_TAG, "");
296  res = res->next;
297  ++i;
298  }
299  payload = payload->next;
300  }
301 }

312 INLINE_API void OCPayloadLogSecurity(LogLevel level, OCSecurityPayload* payload)
313 {
314  size_t payloadSize = payload->payloadSize;
315  OIC_LOG(level, PL_TAG, "Payload Type: Security");
316 
317  if (payloadSize > 0)
318  {
319  // Add a zero-character string terminator.
320  char *securityData = (char *)OICMalloc(payloadSize + 1);
321 
322  if (securityData)
323  {
324  memcpy(securityData, payload->securityData, payloadSize);
325  // assert(securityData[payloadSize - 1] != '\0');
326  securityData[payloadSize] = '\0';
327  OIC_LOG_V(level, PL_TAG, "\tSecurity Data: %s", securityData);
328  OICFree(securityData);
329  }
330  }

333 INLINE_API void OCPayloadLog(LogLevel level, OCPayload* payload)
334 {
335  if(!payload)
336  {
337  OIC_LOG(level, PL_TAG, "NULL Payload");
338  return;
339  }
340  switch(payload->type)
341  {
342  case PAYLOAD_TYPE_REPRESENTATION:
343  OCPayloadLogRep(level, (OCRepPayload*)payload);
344  break;
345  case PAYLOAD_TYPE_DISCOVERY:
346  OCPayloadLogDiscovery(level, (OCDiscoveryPayload*)payload);
347  break;
348  case PAYLOAD_TYPE_PRESENCE:
349  OCPayloadLogPresence(level, (OCPresencePayload*)payload);
350  break;
351  case PAYLOAD_TYPE_SECURITY:
352  OCPayloadLogSecurity(level, (OCSecurityPayload*)payload);
353  break;
354  default:
355  OIC_LOG_V(level, PL_TAG, "Unknown Payload Type: %d", payload->type);
356  break;
357  }
